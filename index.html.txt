function renderTabs(){
  document.getElementById('tabs').innerHTML=tabs.map(t=>`
    <button class="tab-btn ${t.id===activeTab?'active':''}" onclick="switchTab('${t.id}')">${t.icon} ${t.label}</button>`).join('');

function switchTab(id){activeTab=id;selected.clear();renderTabs();applyFilters();}
function setRoom(r){
  activeRoom=r;
  document.querySelectorAll('.room-btn').forEach(b=>b.classList.toggle('active',b.textContent.trim()===(r||'전체')));
  applyFilters();
}
function getFiltered(){
  const tab=tabs.find(t=>t.id===activeTab);if(!tab)return[];
  const s=document.getElementById('search').value.toLowerCase();
  return tab.courses.filter(c=>{
    if(activeRoom&&c.room!==activeRoom)return false;
    if(s&&!c.subject.toLowerCase().includes(s))return false;
    return true;
  });
}
function applyFilters(){
  const f=getFiltered();
  document.getElementById('stats-text').innerHTML=`총 <b>${f.length}</b>개 과목`;
  renderList(f);updateSelInfo();
}
function renderList(courses){
  const el=document.getElementById('list');
  if(!courses.length){el.innerHTML=`<div class="empty"><span class="empty-icon">📭</span><div class="empty-title">강좌가 없습니다</div></div>`;return;}
  el.innerHTML=courses.map(c=>`
    <div class="list-item ${selected.has(c.id)?'selected':''}" id="item-${c.id}" onclick="onItemClick(${c.id})">
      <input type="checkbox" class="list-chk ${multiMode?'show':''}" id="chk-${c.id}" ${selected.has(c.id)?'checked':''} onclick="event.stopPropagation()" onchange="toggleSel(${c.id},this.checked)">
      <span class="room-badge">${c.room}</span>
      <div class="item-info">
        <div class="item-name">${c.subject}</div>
        <div class="item-meta">
          <span>${c.days}</span>
          <span>${c.timeStart}~${c.timeEnd}</span>
          <span>${c.startDate}~${c.endDate}</span>
          ${c.teacher?`<span>${c.teacher}</span>`:''}
        </div>
      </div>
      <button class="copy-btn" id="cb-${c.id}" onclick="copyItem(${c.id},event)">📋 복사</button>
    </div>`).join('');
}
function onItemClick(id){
  if(!multiMode)return;
  const chk=document.getElementById('chk-'+id);
  if(chk){chk.checked=!chk.checked;toggleSel(id,chk.checked);}
}
function toggleSel(id,checked){
  if(checked)selected.add(id);else selected.delete(id);
  const item=document.getElementById('item-'+id);
  if(item)item.classList.toggle('selected',selected.has(id));
  updateSelInfo();
}
function updateSelInfo(){
  const el=document.getElementById('sel-count');
  el.textContent=selected.size+'개 선택됨';
  el.classList.toggle('active',selected.size>0);
  const show=multiMode&&selected.size>0;
  document.getElementById('sel-copy-btn').style.display=show?'':'none';
  document.getElementById('clear-btn').style.display=multiMode?'':'none';
}
function toggleMulti(){
  multiMode=!multiMode;
  document.getElementById('multi-btn').classList.toggle('on',multiMode);
  if(!multiMode)selected.clear();
  applyFilters();updateSelInfo();
}
function clearSel(){selected.clear();applyFilters();updateSelInfo();}

function copyItem(id,e){
  e.stopPropagation();
  const tab=tabs.find(t=>t.id===activeTab);const c=tab&&tab.courses.find(c=>c.id===id);if(!c)return;
  navigator.clipboard.writeText(genMsg([c])).then(()=>{
    const btn=document.getElementById('cb-'+id);
    if(btn){btn.classList.add('copied');btn.textContent='✓ 복사됨';setTimeout(()=>{btn.classList.remove('copied');btn.textContent='📋 복사';},1800);}
    toast('✅ 복사되었습니다!');
  });
}

  const f=getFiltered();if(!f.length){toast('강좌가 없습니다.');return;}
  navigator.clipboard.writeText(genMsg(f)).then(()=>toast(`✅ 전체 ${f.length}개 복사 완료!`));
}
function copySelected(){
  const tab=tabs.find(t=>t.id===activeTab);if(!tab)return;
  const toGen=tab.courses.filter(c=>selected.has(c.id));if(!toGen.length){toast('선택된 강좌가 없습니다.');return;}
  navigator.clipboard.writeText(genMsg(toGen)).then(()=>{
    toast(`✅ ${toGen.length}개 선택 복사 완료!`);
    // 복사 후 다중선택 초기화
    multiMode=false;
    selected.clear();
    document.getElementById('multi-btn').classList.remove('on');
    applyFilters();updateSelInfo();
  });
}
function handleFile(file){
  if(!file)return;setStatus('⏳ 읽는 중...','');
  const r=new FileReader();
  r.onload=e=>{
    try{const text=new TextDecoder('utf-8').decode(new Uint8Array(e.target.result));
      if(text.includes('전체출석율'))parseSBS(text,file.name);
      else setStatus('❌ 지원하지 않는 형식입니다.','err');
    }catch(err){setStatus('❌ 오류: '+err.message,'err');}
  };r.readAsArrayBuffer(file);
}
function parseSBS(html,filename){
  const rows=[...html.matchAll(/<tr[^>]*>(.*?)<\/tr>/gs)].map(m=>m[1]);
  const col2room=['TIME'];
  const thTags=[...(rows[0]||'').matchAll(/<th([^>]*)>(.*?)<\/th>/gs)];
  thTags.forEach(([,attrs,text])=>{
    const clean=text.replace(/<[^>]+>/g,'').trim();
    const cs=parseInt((attrs.match(/colspan=["']?(\d+)/)||[0,1])[1]);
    if(clean!=='강의실')for(let i=0;i<cs;i++)col2room.push(clean+'강의장');
  });
  const occ={},timeAt={},raw=[];
  rows.forEach((rh,ri)=>{
    if(ri<3)return;const thM=rh.match(/<th[^>]*>(\d+:\d+)<\/th>/);if(!thM)return;
    timeAt[ri]=thM[1];let col=1;
    [...rh.matchAll(/<td([^>]*)>(.*?)<\/td>/gs)].forEach(([,attrs,ch])=>{
      while(occ[ri+','+col])col++;
      const text=ch.replace(/&nbsp;/g,' ').replace(/<[^>]+>/g,' ').replace(/\s+/g,' ').trim();
      const rs=parseInt((attrs.match(/rowspan=["']?(\d+)/)||[0,1])[1]);
      if(text.includes('전체출석율')&&col<col2room.length)raw.push({ri,col,text,rs});
      for(let r=ri+1;r<ri+rs;r++)occ[r+','+col]=true;col++;
    });
  });
  const fmt=d=>{const p=d.split('-');return p[1]+'.'+p[2];};
  const parsed=raw.map(({ri,col,text,rs})=>{
    const subject=text.split('전체출석율')[0].trim();
    const tm=text.match(/배정:\d+(?:\([^)]+\))?\s+([가-힣]+)\d*\s+[월화수목금토일~\/]+\s+개:/);
    const teacher=tm?tm[1]+'강사':'';
    const dm=text.match(/배정:\d+(?:\([^)]+\))?\s+(?:[가-힣]+\d*\s+)?([월화수목금토일~\/]+)\s+개:/);
    const days=dm?dm[1].trim():'';
    const sm=text.match(/개:(\d{4}-\d{2}-\d{2})/);const em=text.match(/종:(\d{4}-\d{2}-\d{2})/);
    const st=timeAt[ri]||'00:00';const[sh,smn]=st.split(':').map(Number);
    const tot=sh*60+smn+rs*30;
    const et=String(Math.floor(tot/60)).padStart(2,'0')+':'+String(tot%60).padStart(2,'0');
    const isWe=/토|일/.test(days);
    return{id:nextId++,subject,teacher,days,isWeekend:isWe,startDate:sm?fmt(sm[1]):'',endDate:em?fmt(em[1]):'',timeStart:st,timeEnd:et,room:col2room[col]||'?',supplies:'X',isOnline:'대면'};
  });
  const mon=[...new Set(parsed.map(c=>c.startDate.split('.')[0]).filter(Boolean))];
  const ml=mon.length?mon[0]+'월':'?월';
  const wd=parsed.filter(c=>!c.isWeekend),we=parsed.filter(c=>c.isWeekend);
  const wdId='wd_'+ml,weId='we_'+ml;
  tabs=tabs.filter(t=>t.id!==wdId&&t.id!==weId);
  if(wd.length)tabs.push({id:wdId,label:ml+' 평일반',icon:'📅',courses:wd});
  if(we.length)tabs.push({id:weId,label:ml+' 주말반',icon:'🗓️',courses:we});
  activeTab=tabs[0]?.id||activeTab;selected.clear();renderTabs();applyFilters();
  setStatus(`✅ "${filename}" — 평일 ${wd.length}개 / 주말 ${we.length}개 로드 완료`,'ok');
  toast(`✅ ${parsed.length}개 강좌 불러오기 완료!`);
}
